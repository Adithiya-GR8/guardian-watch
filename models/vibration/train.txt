def extract_signal_from_file(file_path):
    try:
        data = loadmat(file_path)
        key = [k for k in data.keys() if not k.startswith("__")][0]
        obj = data[key][0][0]

        Y = obj['Y'][0]
        for item in Y:
            try:
                for element in item:
                    if isinstance(element, np.ndarray) and element.size > 1000:
                        return element.flatten()
            except:
                continue
    except Exception as e:
        print(f"Error in {file_path}: {e}")
    return None

def load_all_mat_files(folder_path, max_files=5):
    all_signals = []
    files = sorted([f for f in os.listdir(folder_path) if f.endswith(".mat")])
    for file in files[:max_files]:
        file_path = os.path.join(folder_path, file)
        signal = extract_signal_from_file(file_path)
        if signal is not None:
            all_signals.append(signal)
            print(f"✅ Loaded {file}")
        else:
            print(f"⚠️ Could not extract signal from {file}")
    if len(all_signals) == 0:
        raise ValueError("❌ No valid signals extracted.")
    return np.concatenate(all_signals)

WINDOW_SIZE = 50
STEP_SIZE = 10

def create_features(data):
    windows = [data[i:i+WINDOW_SIZE] for i in range(0, len(data) - WINDOW_SIZE, STEP_SIZE)]
    # Ensure there are windows to create features from
    if not windows:
        return np.array([])
    # Features: mean, std, max, min, RMS
    feats = [[np.mean(w), np.std(w), np.max(w), np.min(w), np.sqrt(np.mean(w**2))] for w in windows]
    return np.array(feats)

X_features = create_features(signal)

# Handle cases where X_features might be empty
if X_features.size == 0:
    raise ValueError("❌ No features created. Check signal length and window/step sizes.")

X = X_features.reshape(X_features.shape[0], 1, X_features.shape[1])

# ==========================================
# AUTOENCODER MODEL (FIXED ARCHITECTURE)
# ==========================================
num_features = X.shape[2]
inputs = Input(shape=(1, num_features))
encoded = LSTM(16, activation='relu')(inputs)
decoded = RepeatVector(1)(encoded)
decoded = LSTM(16, activation='relu', return_sequences=True)(decoded)
# Added Dense layer to match the input feature dimension (5)
outputs = Dense(num_features)(decoded)

model = Model(inputs, outputs)
model.compile(optimizer='adam', loss='mse')

# Calculate reconstruction errors on the training data to determine threshold

X_reshaped = X.reshape(X.shape[0], X.shape[2])

train_errors = np.mean(np.abs(X_reshaped - X_pred_reshaped), axis=1)

# Function to detect anomalies in new data
def detect_anomalies(new_data_signal, trained_scaler, trained_model, window_size, step_size, anomaly_threshold):
    # Scale the new signal using the trained scaler
    scaled_new_signal = trained_scaler.transform(new_data_signal.reshape(-1, 1)).flatten()

    # Create features from the scaled signal
    new_X_features = create_features(scaled_new_signal)

    if new_X_features.size == 0:
        print("⚠️ Not enough data to create features for anomaly detection.")
        return [], []

    new_X = new_X_features.reshape(new_X_features.shape[0], 1, new_X_features.shape[1])

    # Predict (reconstruct) the new features
    new_X_pred = trained_model.predict(new_X)

    # Reshape for error calculation
    new_X_reshaped = new_X.reshape(new_X.shape[0], new_X.shape[2])
    new_X_pred_reshaped = new_X_pred.reshape(new_X_pred.shape[0], new_X_pred.shape[2])

    # Calculate reconstruction errors
    new_errors = np.mean(np.abs(new_X_reshaped - new_X_pred_reshaped), axis=1)

    # Identify anomalies
    anomalies = new_errors > anomaly_threshold
    return new_errors, anomalies

# Let's instead show how a *function* for real-time prediction would work:
def predict_critical_vibration(latest_signal_window_raw, trained_scaler, trained_model, anomaly_threshold):
    """
    Predicts if the latest vibration data indicates a critical state.
    Args:
        latest_signal_window_raw (np.ndarray): A raw, unscaled vibration signal window.
        trained_scaler (MinMaxScaler): The scaler fitted on training data.
        trained_model (tf.keras.Model): The trained autoencoder model.
        anomaly_threshold (float): The calculated anomaly threshold.
    Returns:
        bool: True if critical, False otherwise.
        float: The reconstruction error for the given window.
    """
    if len(latest_signal_window_raw) < WINDOW_SIZE:
        print("Error: Input signal window is too short to create features.")
        return False, 0.0

    # Scale the input signal window
    scaled_window = trained_scaler.transform(latest_signal_window_raw.reshape(-1, 1)).flatten()

    # Create features for the single window
    # Note: create_features expects a longer data sequence to generate multiple windows.
    # For a single window, we need to adapt it.
    # We will just process the one window directly as features.
    feats_for_prediction = np.array([
        np.mean(scaled_window),
        np.std(scaled_window),
        np.max(scaled_window),
        np.min(scaled_window),
        np.sqrt(np.mean(scaled_window**2))
    ]).reshape(1, 1, num_features) # Reshape to (1, 1, num_features) for the model

    # Make prediction (reconstruction)
    reconstruction = trained_model.predict(feats_for_prediction)

    # Calculate reconstruction error
    error = np.mean(np.abs(feats_for_prediction.reshape(num_features) - reconstruction.reshape(num_features)))

    is_critical = error > anomaly_threshold
    return is_critical, error
